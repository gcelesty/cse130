# pseudocode program

function find_primes_below_n(n):
	# create an array of Boolean values, initially true
	is_prime = [True] * (n + 1)

	# cross out multiple of 2
	for i from 4 to n step 2:
		is_prime[i] = False

	# cross out multiples of 3
	for i from 6 to n step 3:
		is_prime[i] = False

	# iterate through odd numbers starting from 5
	for i from 4 to sqrt(n) + 1 step 2:
		if is_prime[i]:
			# cross out multiples of i
			for j from i*i to n step i:
				is_prime[j] = False

	# create a list to store prime numbers
	primes = []

	# add primes to the list
	for i from 2 to n:
		if is_prime[i]:
			append i to primes

	# display the primes
	display primes
